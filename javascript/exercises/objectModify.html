Overwrite Object Properties

Change the cities listed in the UnitedStatesCities object to the each state's capitol. Use console.log statements to check your work. 

   var UnitedStatesCities = {
      NewYork : "New York City",
      California: "Los Angeles",
      Illinois: "Chicago",
      Texas: "Houston",
      Pennsylvania: "Philadelphia",
      Arizona: "Phoenix"
   }


UnitedStatesCities["NewYork"] = "Albany"
UnitedStatesCities.California = "Sacramento"
UnitedStatesCities.Illinois = "Springfield"
UnitedStatesCities["Texas"] = "Austin"
UnitedStatesCities["Pennsylvania"] = "Harrisburg"
UnitedStatesCities.Arizona = "Phoenix"


console.log(UnitedStatesCities.NewYork); // "New York City"
console.log(UnitedStatesCities.California); // "Los Angeles"
console.log(UnitedStatesCities.Illinois); // "Chicago"
console.log(UnitedStatesCities["Texas"]); // "Houston"
console.log(UnitedStatesCities["Pennsylvania"]); // "Philadelphia"
console.log(UnitedStatesCities["Arizona"]); // "Phoenix"
